## 引入Vue

<!--more-->

创建Vue对象`v`，`v`挂载到页面的id为root的div元素上。挂载除了调用`$mount`，也可以通过指定`el`属性操作

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <div id="root">
    </div>
    <script src="../js/vue.js"></script>
    <script>
        Vue.config.productionTip = false;
        const v = new Vue({
            data: function() {
                return {};
            },
            methods: {
            },
            computed: {
            },
            watch: {
            }
        });
        v.$mount('#root');
    </script>
</body>
</html>
```



## 数据绑定

```html
<div id="root">
    <h1>Hello,{{name}}</h1>
    <hr/>
    <h1>{{school.name}}</h1>
    <a v-bind:href="school.url">MIT Website</a>
    <hr/>
    <input type="text" name="address" id="address" v-model:value="school.address">
    <div>
        <input type="text" name="first" id="first" v-model:value="related.first">
    </div>

    <div>
        <input type="text" name="second" id="first" v-model:value="related.second">
    </div>
</div>
```

```js
const v = new Vue({
    data: function() {
        return {
            name: 'Clio',
            school: {
                name: 'MIT',
                url: 'https://www.bilibili.com/',
                address: 'America'
            },
            related: {
                first: '1',
                second: '2'
            }
        };
    }
});
```

简写语法

```html
<!-- 单向绑定 -->
<a v-bind:href="school.url">MIT Website</a>
<a :href="school.url">MIT Website</a>

<!-- 双向绑定 -->
<input type="text" name="first" id="first" v-model:value="related.first">
<input type="text" name="first" id="first" v-model="related.first">

```

## 计算函数

```html
<div id="root">
    <div>
        {{concat}}
    </div>
</div>
```

```js
const v = new Vue({
    data: function() {
        return {
            related: {
                first: '1',
                second: '2'
            }
        };
    },
    computed: {
        concat: function() {
            return this.related.first + '-' + this.related.second;
        }
    }
});
```



## 监视函数

可以监视`data`的属性和`computed`方法计算的结果值

```js
const v = new Vue({
    watch: {
        computedSchool: {
            handler: function(newValue, oldValue) {
                console.log(oldValue, '->', newValue)
            }
        },
        'innerNumber.a': {
            handler: function(newValue, oldValue) {
                console.log('a被改变', oldValue, '->', newValue);
            }
        },
        innerNumber: {
            deep: true,
            handler: function(newValue, oldValue) {
                console.log('innerNumber被改变', oldValue, '->', newValue);
            }
        }
    }
});
```



## 单文件组件

一般不使用

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Hello, Vue!</title>
</head>
<body>
    <div id="root">
        <school></school>
        <student></student>
    </div>

    <script src="vue.js"></script>
    <script>
        Vue.config.productionTip = false;

        const school = Vue.extend({
            template: `<h3>{{name}}</h3>`,
            data() {
                return {
                    name: 'MIT'
                }
            },
        });

        const student = Vue.extend({
            template: `<h3>{{name}}</h3>`,
            data() {
                return {
                    name: 'Clio'
                }
            },
        });

        const vm = new Vue({
            el: '#root',
            components: {
                school,
                student
            }
        });
    </script>
</body>
</html>
```
